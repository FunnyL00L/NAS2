export default function App() {

        </TouchableOpacity>

        <TouchableOpacity

          <Settings color={activeTab === 'settings' ? '#09090b' : '#a1a1aa'} size={24} />
          <Text style={[styles.tabLabel, activeTab === 'settings' && styles.tabLabelActive]}>Settings</Text>

}
